        -	@heroicons/react: 2.0.18,
        -	@tailwindcss/forms: 0.5.6,
        -	@testing-library/jest-dom: 5.17.0,
        -	@testing-library/react: 13.4.0,
        -	@testing-library/user-event: 13.5.0,
        -	axios: 1.6.0,
        -	react: 17.0.0,

        -	react-scripts: 5.0.1,

## Installation

Install front_client with npm

```bash
  npm install
  cd front_client
```
